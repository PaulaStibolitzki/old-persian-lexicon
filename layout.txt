function searchWord(){
let query = document.getElementById("searchBox").value.toLowerCase()
let mode = document.getElementById("searchMode").value
let posFilter = document.getElementById("posFilter").value
let genderFilter = document.getElementById("genderFilter").value

let results = []

for(let id in data.entries){

let e = data.entries[id]
let match = true

if(query !== ""){

if(mode === "lemma"){
match =
(e.lemma && e.lemma.toLowerCase().startsWith(query)) ||
(e.variants && e.variants.some(v => v.toLowerCase().startsWith(query)))}

else if(mode === "definition"){
match = e.definition && e.definition.toLowerCase().includes(query)}

else{
match =
(e.lemma && e.lemma.toLowerCase().startsWith(query)) ||
(e.definition && e.definition.toLowerCase().includes(query))}

}if(!match) continue

if(posFilter && e.pos !== posFilter) continue
if(genderFilter && e.gender !== genderFilter) continue

results.push(e)

}displayResults(results)}
